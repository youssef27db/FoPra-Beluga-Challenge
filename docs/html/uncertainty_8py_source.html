9def add_reference_arrivals(prb : BelugaProblem,
10 seed : int,
11 prec : int = 2):
12 # Load rates from the included file
13 rates_file = resources.files(configuration) / 'hourly_rates.json'
14 with rates_file.open() as fp:
15 rates = json.load(fp)
16 rates = {int(k) : v for k, v in rates.items()}
17 # Build and seed an RNG
18 rng = np.random.default_rng(seed)
19 # Prepare all flights for processing
20 queue = prb.flights.copy()
21 # flight_times = []
22 now = 0
23 period = max(rates.keys()) + 1
24 while len(queue) > 0:
25 # Draw a number of flights for this time
26 n_flight_per_step = rng.poisson(rates[now % period])
27 # Draw arrivals times uniformly within the hour
28 offsets = rng.uniform(size=n_flight_per_step)
29 offsets = np.round(offsets, decimals=prec)
30 # Sort these local arrival times
31 offsets = np.sort(offsets)
32 for k, offset in enumerate(offsets):
33 # Assign the scheduled arrival time
34 queue[0].scheduled_arrival = now + offset
35 # Remove the flight from the queue
36 queue.pop(0)
37 # Early stop
38 if len(queue) == 0:
39 break
40 # Advance time
41 now += 1

73 def sample_arrivals_times(self,
74 scheduled_arrivals : Iterable[float],
75 size : int = None,
76 seed : int = None,
77 rebase : bool = True):
78 assert(size is None or size > 0)
79 # Build and seed an RNG
80 rng = np.random.default_rng(seed)
81 # Convert scheduled_arrivals to an array
82 scheduled_arrivals = np.array(scheduled_arrivals)
83 # Determine the number of samples
84 nsamples = 1 if size is None else size
85 # Sample delay vectors
86 delays = rng.choice(self.support,
87 size=(nsamples, scheduled_arrivals.shape[0]),
88 p=self.probsprobs)
89 # Combine delays and scheduled_arrivals
90 times = scheduled_arrivals + delays
91 # Rebase, if requested
92 if rebase:
93 new_bases = np.min(times, axis=1).reshape(-1, 1)
94 new_bases = np.floor(new_bases / 24) * 24
95 times -= new_bases
96 # Return the result
97 if size is None:
98 return times.ravel()
99 else:
100 return times

156def _cartesian_product(arrays):
157 la = len(arrays)
158 dtype = np.result_type(*arrays)
159 arr = np.empty([len(a) for a in arrays] + [la], dtype=dtype)
160 for i, a in enumerate(np.ix_(*arrays)):
161 arr[...,i] = a
162 return arr.reshape(-1, la)

164def _remove_replicated(cprod):
165 mask = [np.unique(a, return_counts=True)[1].max() == 1 for a in cprod]

169def _build_tt_support(nflights, history_len):
170 # Build the cartesian product of all flights
171 flight_choices = np.arange(nflights)
172 transition_support = _cartesian_product([flight_choices] * (history_len+1))
173 transition_support = _remove_replicated(transition_support)
174 # Add starting transitions
175 for k in range(history_len):
176 tail = _cartesian_product([flight_choices] * (history_len-k))
177 tail = _remove_replicated(tail)
178 head = np.full(shape=(tail.shape[0], k+1), fill_value=-1)
179 ext = np.hstack((head, tail))
180 transition_support = np.vstack((ext, transition_support))
181 # # Define the state support
182 # state_support = np.unique(transition_support[:, :-1], axis=0)
183 return set(tuple(t) for t in transition_support)

186def _count_transitions(prb, history_len, nsamples, seed):
187 # Sample multiple scenarios (i.e. flight arrival sequences)
188 sampler = ArrivalSampler()
189 sampler.setup()
190 scheduled_arrivals = [f.scheduled_arrival for f in prb.flights]
191 scenarios, _ = sampler.sample_id_sequences(scheduled_arrivals=scheduled_arrivals,
192 size=nsamples, seed=seed, rebase=False)
193 # Add starting elements
194 base = np.full(shape=(scenarios.shape[0], history_len), fill_value=-1)
195 scenarios = np.hstack((base, scenarios))
196 # Build 3d tensor representation of the sequences
197 # Subsequences include the state (history_len flights), plus the
198 # next flight
199 slen = scenarios.shape[1]
200 subsequences = np.stack([
201 scenarios[:, k:slen-(history_len-k)] for k in range(history_len+1)
202 ], axis=-1)
203 # Count the occurrences of each subsequence
204 flat_subsequences = subsequences.reshape(subsequences.shape[0] * subsequences.shape[1], history_len+1)
205 seqs, counts = np.unique(flat_subsequences, axis=0, return_counts=True)
206 cmap = {tuple(k): v for k, v in zip(seqs, counts)}
207 return cmap

216def _normalize_counts(tt_counts, prec):
217 # Compute the normalization constants
218 z_map = {}
219 cur_state = None
220 cur_ccount = 0
221 last_tt = None
222 last_tt_map = {}
223 for tt in sorted(tt_counts):
224 stt = tuple(tt[:-1])
225 cnt = tt_counts[tt]
226 if cur_state is None:
227 cur_state = stt
228 if stt != cur_state:
229 # Othewise, store the prob. mass
230 z_map[cur_state] = cur_ccount
231 # Mark the last transition for such state
232 last_tt_map[cur_state] = last_tt
233 # Then reset the accumulation fields
234 cur_state = stt
235 cur_ccount = 0
236 # Update the count
237 cur_ccount += cnt
238 # Update the last transition
239 last_tt = tt
240 # Store the mass for the last state
241 z_map[cur_state] = cur_ccount
242 last_tt_map[cur_state] = tt
243
244 # Normalize the counts
245 cur_pmass = 0
246 for tt in sorted(tt_counts):
247 stt = tuple(tt[:-1])
248 if tt != last_tt_map[stt]:
249 # If this is not the last transition for this state,
250 # apply a typical normalization
251 val = np.round(tt_counts[tt] / z_map[stt], prec)
252 tt_counts[tt] = val
253 # Update the probability mass
254 cur_pmass += val
255 else:
256 # Otherwise, normalize by complement
257 tt_counts[tt] = np.round(1 - cur_pmass, prec)
258 # Reset the probability mass
259 cur_pmass = 0
260
261 return tt_counts

264def add_abstract_uncertainty_model(prb : BelugaProblem,
265 history_len : int = 1,
266 seed : int = None,
267 nsamples : int = 10000):
268 # Define the support for the transition table
269 nflights = len(prb.flights)
270 tt_support = _build_tt_support(nflights, history_len)
271 # Compute transition counts via Monte-Carlo simulation
272 tt_counts = _count_transitions(prb, history_len, nsamples, seed)
273 # Add the missing probabilities
274 _add_missing_transition(tt_counts, tt_support)
275 # Normalize to obtain probabilities
276 prec = int(np.ceil(np.log10(nsamples)))
277 tt_probs = _normalize_counts(tt_counts, prec)
278 # Write the transition table on the problem object
279 tt_last = []
280 tt_next = []
281 tt_prob = []
282 for tt in tt_probs:
283 tt_last.append([prb.flights[int(v)].name if int(v) >= 0 else None for v in tt[:-1]])
284 tt_next.append(prb.flights[int(tt[-1])].name)
285 tt_prob.append(float(tt_probs[tt]))
286 prb.tt_last = tt_last
287 prb.tt_next = tt_next
288 prb.tt_prob = tt_prob
