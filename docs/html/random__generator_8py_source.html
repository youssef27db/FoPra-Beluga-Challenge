656 if mandatory >= 0 and mandatory not in seq:
657 choices = [mandatory]
658 else:
659 choices = range(max(0, pos - win), min(num, pos + win + 1))
660 if verbose > 0:
661 print(f"seq={seq}, pos={pos}, win={win}, num={num}, choices={choices}")
662 # Consider all possible next nodes
663 visited = set(seq)
664 for k in choices:
665 if k in visited:
666 continue
667 nxt = seq + [k]
668 res.append(tuple(nxt))

670 def _assign_probabilities(tbl, pos, potential, prec=5):
671 # Assign a potential to the transitions
672 pvals = []
673 zvals = {}
674 last_of_group = []
675 prev_last = None
676 for lst, nxt in zip(tbl["last"], tbl["next"]):
677 # Obtain the potential
678 pval = potential(pos, lst, nxt)
679 pvals.append(pval)
680 # Update the normalization factor
681 if lst not in zvals:
682 zvals[lst] = pval
683 else:
684 zvals[lst] += pval
685 # Flag the previous transition if it was the last in a group
686 last_of_group.append(prev_last is not None and lst != prev_last)
687 # The last transition always closes a group
688 last_of_group.append(True)
689 # Normalize the potentials to obtain probabilities
690 tbl["prob"] = []
691 current_group_prob = 0
692 for lst, nxt, pval, log in zip(tbl["last"], tbl["next"], pvals, last_of_group):
693 # Determine the probability and update the current group prob.
694 if not log:
695 prob = np.round(pval / zvals[lst], decimals=prec)
696 current_group_prob += prob
697 else:
698 prob = 1 - current_group_prob
699 current_group_prob = 0
700 tbl["prob"].append(prob)

714 )
715 # Display the local transition table
716 if verbose > 1:

718 for row in tbl:
719 print(row)
720 # Store local transition table
721 tbl = sorted(tbl)
722 itbl.append(tbl)
723 # Build the predecessor table for the next iteration
724 ptbl = set(t[1:] for t in tbl)
725 ptbl = [list(t) for t in ptbl]
726 if verbose > 1:
727 print(f"--- next predecessor table")
728 for row in ptbl:
729 print(row)

731 res[pos] = {"last": [t[:-1] for t in tbl], "next": [t[-1] for t in tbl]}
732 # Assign a probability value to the transitions
733 ProbabilisticModelGenerator._assign_probabilities(res[pos], pos, potential)
734 # Print the transition table
735 if verbose > 0:
736 print(f"--- transition table")
