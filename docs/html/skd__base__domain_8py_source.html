21class State:
22 """Class defining the state type used by the scikit-decide domains"""
23
24 def __init__(self, domain: Any, state: PladoState, cost_function: set[int] = set()):
25 self.atoms: tuple[tuple[tuple[int]]] = tuple(
26 tuple(sorted(state.atoms[p])) for p in range(len(state.atoms))
27 )
28 self.domain: Any = domain
29 self.fluents: tuple[tuple[tuple[int], int]] = tuple(
30 (
31 tuple(
32 (args, int(state.fluents[f][args]))
33 for args in sorted(state.fluents[f].keys())
34 )
35 if f not in cost_function
36 else tuple()
37 )
38 for f in range(len(state.fluents))
39 )
40
41 def to_plado(self, cost_functions: set[int]) -> PladoState:
42 state = PladoState(0, len(self.fluents))
43 state.atoms = self.atoms
44 for f, values in enumerate(self.fluents):
45 for args, val in values:
46 state.fluents[f][args] = Float(val)
47 for f in cost_functions:
48 state.fluents[f][tuple()] = Float(0)
49 return state
50
51 def __str__(self) -> str:
52 return self.domain.task.dump_state(self.to_plado(self.domain.cost_functions))
53
54 def __hash__(self) -> int:
55 return hash((self.atoms, self.fluents))
56
57 def __eq__(self, o: object) -> bool:
58 return (
59 isinstance(o, State) and self.atoms == o.atoms and self.fluents == o.fluents

94class ObservationSpace(Space[D.T_observation]):
95 """Class defining the observation space used by the scikit-decide domains"""
96
97 def __init__(
98 self,
99 predicate_arities: Iterable[int],
100 function_arities: Iterable[int],
101 num_objects: int,
102 ):
103 self.predicate_arities: tuple[int] = tuple(predicate_arities)
104 self.function_arities: tuple[int] = tuple(function_arities)
105 self.num_objects: int = num_objects
106
107 def contains(self, x: D.T_observation) -> bool:
108 if len(x.atoms) != len(self.predicate_arities) or len(x.fluents) != len(
109 self.function_arities
110 ):
111 return False
112 for p in range(x.atoms):
113 for params in x.atoms[p]:
114 if (
115 len(params) != self.predicate_arities[p]
116 or min(params) < 0
117 or max(params) >= self.num_objects
118 ):
119 return False
120 for f in range(x.fluents):
121 for params, _ in x.fluents[f]:
122 if (
123 len(params) != self.function_arities[f]
124 or min(params) < 0
125 or max(params) >= self.num_objects
126 ):
127 return False
128 return True

214 def _init_deserializer(self):

228 def _deserialize_objects(self, objs: tuple[str]) -> tuple[int]:
229 res = []
230 for o in objs:
231 res.append(self._object_idx.get(o, None))
232 if res[-1] is None:
233 raise ValueError(f"object {o} doesn't exist")

236 def _deserialize_atom(self, atom: tuple[str]) -> tuple[int] | None:
237 if len(atom) == 0:
238 raise ValueError("empty tuple doesn't represent a fact")
239 idx = self._predicate_idx.get(atom[0].lower(), None)
240 if idx is None:
241 raise ValueError(f"predicate {atom[0]} doesn't exist")
242 params = self._deserialize_objects(atom[1:])
243 if idx > self.task.num_fluent_predicates:
244 return None
245 return (idx, *params)
246
247 def _deserialize_fluent_atom(self, atom: tuple[str]) -> tuple[int] | None:
248 if len(atom) == 0:
249 raise ValueError("empty tuple doesn't represent a fluent")
250 idx = self._function_idx.get(atom[0].lower(), None)
251 if idx is None:
252 raise ValueError(f"function {atom[0]} doesn't exist")
253 params = self._deserialize_objects(atom[1:])
254 return (idx, *params)

256 def deserialize_state(
257 self, atoms: Iterable[tuple[str]], fluents: Iterable[tuple[tuple[str], int]]
258 ):
259 atoms_ = tuple(
260 x for x in (self._deserialize_atom(atom) for atom in atoms) if x is not None
261 )
262 fluents_ = tuple(
263 (self._deserialize_fluent_atom(atom), value) for (atom, value) in fluents
264 )
265 state = State(self, PladoState(0, 0), [])
266 state.atoms = atoms_
267 state.fluents = fluents_

270 def deserialize_action(self, action: tuple[str]) -> Action:
271 if len(action) == 0:
272 raise ValueError("empty tuple doesn't represent an action")

275 raise ValueError(f"action {action[0]} doesn't exist")
