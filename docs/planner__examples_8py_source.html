23def _generate_pddl(

27 state : BelugaProblemState = None,
28 ):

34 domain_encoding = DomainEncoding(variant, beluga_problem)
35 domain_str = domain_encoding.domain.to_pddl("beluga")

37 problem_name = 'Internal Beluga Problem Instance'
38 pddl_problem = encode(name=problem_name,
39 beluga_problem=beluga_problem,
40 domain=domain_encoding.domain,

42 state=state)

165class RandomDeterministicPlanner(DeterministicPlannerAPI):
166
167 def __init__(self,
168 max_steps : int = 100,
169 classic : bool = False):
170 self.classic = classic
171 self.max_steps = max_steps

174 pass
175
176 def build_plan(self, prb : BelugaProblem):
177 # Build a SKD domain
178 domain = SkdPDDLDomain(prb, problem_name='server_side_domain',
179 classic=self.classic)
180 action_space = domain.get_action_space()
181 observation_space = domain.get_observation_space()
182
183 res = BelugaPlan()
184 s = domain.reset()
185 for step in range(self.max_steps):
186 # Stop the process if the goal has been reached
187 if domain._is_terminal(s):
188 break
189 # Determine applicable actions
190 actions = domain.get_applicable_actions(s)
191 # Stop the process if a dead-end has been reached
192 if len(actions.get_elements()) == 0:
193 break
194 # Sample an applicable action
195 a = actions.sample()
196 # Convert the action in the competition format
197 ba = _skd_action_to_beluga_action(action=a, domain=domain, classic=self.classic)
198 # Store the action in the plan
199 res.append(ba)
200 # Apply the action and move to the next state
201 o = domain.step(a)
202 s = o.observation
203
204 # # Return the plan
205 # return res
206
207 # Cleanup
208 domain.cleanup()
209
210 # Return the result
211 return res
