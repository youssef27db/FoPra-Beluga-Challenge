212def action_from_json_obj(json_obj : dict, prb : BelugaProblem):
213 # Check whether an action
214 if 'name' not in json_obj:
215 raise ValueError('Invalid action, missing "name" field')
216 # Parse the action arguments depending on their type
217 res = None
218 if json_obj['name'] == LoadBeluga.name:
219 res = LoadBeluga.from_json_obj(json_obj, prb)
220 elif json_obj['name'] == UnloadBeluga.name:
221 res = UnloadBeluga.from_json_obj(json_obj, prb)
222 elif json_obj['name'] == GetFromHangar.name:
223 res = GetFromHangar.from_json_obj(json_obj, prb)
224 elif json_obj['name'] == DeliverToHangar.name:
225 res = DeliverToHangar.from_json_obj(json_obj, prb)
226 elif json_obj['name'] == PutDownRack.name:
227 res = PutDownRack.from_json_obj(json_obj, prb)
228 elif json_obj['name'] == PickUpRack.name:
229 res = PickUpRack.from_json_obj(json_obj, prb)
230 elif json_obj['name'] == SwitchToNextBeluga.name:
231 res = SwitchToNextBeluga.from_json_obj(json_obj, prb)
232 else:
233 raise ValueError('Invalid action name "{}"'.format(json_obj['name']))
234 return res

237class BelugaPlan:
238
239 def __init__(self, actions=None):
240 if actions is None:
241 self.actions = []
242 else:
243 self.actions = actions
244
245 def append(self, action : BelugaAction):
246 self.actions.append(action)
247
248 def to_json_obj(self):
249 return [a.to_json_obj() if a is not None else None for a in self.actions]
250
251 def to_json_str(self):
252 return json.dumps(self.to_json_obj())
253
254 def __repr__(self):
255 return '\n'.join(a.to_json_str() for a in self.actions)
256 # return self.to_json_str()
257
258 def from_json_obj(json_obj, prb: BelugaProblem):
259 actions = [action_from_json_obj(o, prb) for o in json_obj]
260 return BelugaPlan(actions)
