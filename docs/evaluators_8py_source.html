172def compute_score_dict(outcome : SingleSimulationOutcome,
173 prb : BelugaProblem,
174 alpha : float,
175 beta : float):
176 # First input term (goal reached)
177 A = outcome.goal_reached
178 # Second input term (relative plan length)
179 B = 0
180 if outcome.plan is not None and prb is not None:
181 B = len(outcome.plan.actions) / len(prb.jigs)
182 # Third input term (inverse of the relative number of free racks)
183 C = 0
184 if outcome.free_racks is not None and prb is not None:
185 C = len(prb.racks) / (1 + outcome.free_racks)
186 value = A * np.exp(- alpha * B - beta * C)
187 # res = {'value': value, 'alpha': alpha, 'beta': beta}
188 res = {'value': value, 'alpha': alpha, 'beta': beta,
189 'A': A, 'B': B, 'C': C}
190 return res

281 def _reconstruct_rack_content(self, rname, content, clear_jigs, next_jig):
282 # Handle empty racks
283 if len(content) == 0:
284 return []
285 # Find the starting jig
286 head = None
287 for jig in content:
288 if jig in clear_jigs:
289 head = jig
290 if head is None:
291 raise EvaluationException(f'Cannot identify the starting jig for rack {rname}')
292 # Sort the jigs
293 sorted_jigs = [head]
294 cur_jig = head
295 while cur_jig in next_jig:
296 sorted_jigs.append(next_jig[cur_jig])
297 cur_jig = next_jig[cur_jig]
298 if len(set(sorted_jigs) - content) > 0 or len(content - set(sorted_jigs)) > 0:
299 raise EvaluationException(f'"in" predicates are inconsistent with "next" predicates for {rname}')
300 # Return the result
301 return sorted_jigs

405 elif action_name == 'stack-rack' and ba.name == PutDownRack.name :
406 jig, _, trailer, rack, side, _ = action_args

409 elif action_name == 'pick-up-rack' and ba.name == PickUpRack.name:
410 jig, trailer, rack, side, _ = action_args

695 def _run_simulation(self):
696 # Tell the planner that another episode is starting
697 try:
698 self.planner.setup_episode()
699 except EvaluationException as e:
700 res = SingleSimulationOutcome(plan_construction_time=None,
701 error_msg=f'Error while setting up an episode: {e.args[0]}',
702 plan=None,
703 final_state=None,
704 final_step=None,
705 goal_reached=False,
706 abrupt_plan_end=False,
707 invalid_plan=False,

711 prb=self.prb,
712 alpha=self.alpha,
713 beta=self.beta)

715
716 # Reset the domain state
717 state = self.domain.reset() # This regenerates a PDDL file
718 self.es.refresh_cache() # ...And therefore all translation maps need to be reloaded
719
720 # State fields that cannot be computed based on predicates
721 beluga_seq = []
722 trailer_location = {}
723
724 # Prepare to collect time statistics
725 elapsed_time = 0
726 goal_reached = False
727 time_limit_reached = False
728
729 # Build a plan dynamically
730 plan = BelugaPlan()
731 for current_step in range(self.max_steps):
732 try:
733 # Determine the currently processed flight
734 cbeluga = self.es._get_current_beluga(state)
735 if len(beluga_seq) == 0 or beluga_seq[-1] != cbeluga:
736 beluga_seq.append(cbeluga)
737
738 # Convert the state
739 bstate = self.es._skd_state_to_beluga_state(state, beluga_seq, trailer_location)
740
741 # Obtain the current metadata
742 metadata = ProbabilisticPlanningMetatada(current_step, elapsed_time)
743
744 # DEBUG plot the current state
745 # print('=' * 78)
746 # print('METADATA')
747 # print(metadata.to_json_str())
748 # print('-' * 78)
749 # print('EVALUATOR STATE')
750 # print(state)
751 # print('-' * 78)
752 # print('EVALUATOR CONVERTED STATE')
753 # for k, v in bstate.to_json_obj().items():
754 # print(f'"{k}": {v}')
755
756 # Check whether the goal has been reached
757 if self.domain._is_terminal(state):
758 goal_reached = True
759 break
760
761 # Retrive current action
762 start_time = time.time()
763 ba = self.planner.next_action(bstate, metadata)
764 elapsed_time += time.time() - start_time
765
766 # Add the action to the plan
767 plan.actions.append(ba)
768
769 # print('-' * 78)
770 # print('PLANNER ACTION')
771 # print(ba)
772
773 # Determine whether the time limit has been exceeded
774 if self.time_limittime_limit is not None and elapsed_time > self.time_limittime_limit:
775 time_limit_reached = True
776 raise EvaluationException('Time limit exceeded')
777
778 # Determine whether an action was actually returned
779 if ba is None:
780 raise EvaluationException('No action returned by the planner')
781
782 # Attempt to match the BelugaAction on a valid SKD action
783 action = self.es._find_valid_action(ba, state, beluga_seq)
784
785 # Apply the action and move to the next state
786 o = self.domain.step(action)
787 state = o.observation
788
789 # Update the trailer location
790 self.es._update_trailer_location(ba, trailer_location)
791
792 except EvaluationException as e:
793 res = SingleSimulationOutcome(plan_construction_time=elapsed_time,
794 error_msg=e.args[0],
795 plan=plan,
796 final_state=bstate,
797 final_step=current_step,
798 goal_reached=goal_reached,
799 abrupt_plan_end=False,
800 invalid_plan=False,
801 time_limit_reached=time_limit_reached,
802 step_limit_reached=False,
803 free_racks=self.es._get_free_racks(bstate),
804 prb=self.prb,
805 alpha=self.alpha,
806 beta=self.beta)
807 return res

809 # Return the result of the simulation
810 res = SingleSimulationOutcome(plan_construction_time=elapsed_time,
811 error_msg=None,
812 plan=plan,
813 final_state=bstate,
814 final_step=current_step,
815 goal_reached=goal_reached,
816 abrupt_plan_end=False,
817 invalid_plan=False,
818 time_limit_reached=time_limit_reached,
819 step_limit_reached=(current_step == self.max_steps-1),
820 free_racks=self.es._get_free_racks(bstate),
821 prb=self.prb,
822 alpha=self.alpha,
823 beta=self.beta)
824 return res
